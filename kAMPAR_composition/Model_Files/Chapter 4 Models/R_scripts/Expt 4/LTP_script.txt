

LTP_All12 <- read.csv("LTP_All12.csv")
LTP_All23 <- read.csv("LTP_All23.csv")
LTP_equal <- read.csv("LTP_equal.csv")
LTP_most12 <- read.csv("LTP_most12.csv")
LTP_most23 <- read.csv("LTP_most23.csv")

Syn_Receptors <- as.data.frame(
LTP)


write.csv(Syn_Receptors, "LTP.csv")
